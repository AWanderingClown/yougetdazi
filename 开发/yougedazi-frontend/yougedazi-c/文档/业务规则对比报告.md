- **当前行为**：时间选择器打开时没有预选默认值

**修复建议**：
在 `order-create.js` 的 onLoad 中添加默认时间初始化：
```javascript
onLoad(options) {
  // ... 现有代码 ...
  
  // 设置默认时间（当前时间+1小时，向上取整到30分钟）
  this.setDefaultTime();
},

setDefaultTime() {
  const now = new Date();
  // 加1小时
  now.setHours(now.getHours() + 1);
  // 向上取整到30分钟
  const minutes = now.getMinutes();
  if (minutes > 30) {
    now.setMinutes(0);
    now.setHours(now.getHours() + 1);
  } else if (minutes > 0) {
    now.setMinutes(30);
  }
  
  const month = now.getMonth() + 1;
  const day = now.getDate();
  const hour = String(now.getHours()).padStart(2, '0');
  const minute = String(now.getMinutes()).padStart(2, '0');
  
  this.setData({
    selectedDate: `${month}月${day}日`,
    selectedTime: `${hour}:${minute}-${minute === '00' ? hour : String(now.getHours() + 1).padStart(2, '0')}:${minute === '00' ? '30' : '00'}`
  });
}
```
